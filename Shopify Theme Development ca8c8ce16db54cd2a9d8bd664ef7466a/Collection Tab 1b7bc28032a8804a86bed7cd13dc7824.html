<script>
  document.addEventListener('DOMContentLoaded', function () {
  const tabHeaders = document.querySelectorAll('.tab__headers-{{ id }} .tab__item-{{ id }}');
  const tabContents = document.querySelectorAll('.tab__contents-{{ id }} [data-tab-content]');

  tabHeaders.forEach((header) => {
    header.addEventListener('click', () => {
      const targetTab = header.getAttribute('data-tab');

      tabHeaders.forEach((h) => h.classList.remove('active'));
      tabContents.forEach((c) => c.classList.remove('active'));

      header.classList.add('active');
      document.querySelector(`[data-tab-content="${targetTab}"]`).classList.add('active');
    });
  });
  {% for block in section.blocks %}
  document.getElementById('new-feature_collection__tab').addEventListener('mouseover', () => {
      document.getElementById('prev--{{ block.settings.collection.handle }}-{{ block.id }}').style.left = '4px';
      document.getElementById('next-{{ block.settings.collection.handle }}-{{ block.id }}').style.right = '8px';
  });

   document.getElementById('new-feature_collection__tab').addEventListener("mouseleave", () => {
      document.getElementById('prev--{{ block.settings.collection.handle }}-{{ block.id }}').style.left = '-500px';
      document.getElementById('next-{{ block.settings.collection.handle }}-{{ block.id }}').style.right = '-500px';
   });                       
  {% endfor %}
});
</script>
