<script>
  let loginSec = document.querySelector(".login-{{ id }}")
  let resetSec = document.querySelector(".reset-{{ id }}")
  if(location.href === "https://emani.com/account/login#recover") {
    loginSec.classList.add("hidden")
    resetSec.classList.remove("hidden")
  } else {
    loginSec.classList.remove("hidden")
    resetSec.classList.add("hidden")
  }
  document.querySelector(".forgot--btn").addEventListener("click", e => {
    e.preventDefault()
    location.replace("/account/login#recover");
    loginSec.classList.add("hidden")
    resetSec.classList.remove("hidden")
  })
</script>
